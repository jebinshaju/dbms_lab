CREATE TABLE EMPLOYEE(ENO varchar(5) PRIMARY KEY,ENAME varchar(20) NOT NULL,SALARY varchar(20) NOT NULL,FOREIGN KEY(DNO) REFERENCES DEPARTMENT(DNO),MNGRNO varchar(5),DOJ DATE,JOB varchar(10),ADDRESS varchar(20),CITY varchar(10)CHECK(CITY IN 'cochin,bombay,madras,delhi'),PINCODE varchar(6));
